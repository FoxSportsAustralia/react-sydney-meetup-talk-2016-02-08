Widget.prototype.render = function (data) {
    return renderToHtml(
        <WidgetComponent {...data.view} />,
        'misc-widget',
        {
            data: data.iso,
            settings: this.settings
        }
    );
};
  • data.view is the full payload containing settings business logic
  • data.iso is what is printed on the DOM
  • we request output of react componnent
  • deliver it to the DOM with settings and data to be picked up by the client
