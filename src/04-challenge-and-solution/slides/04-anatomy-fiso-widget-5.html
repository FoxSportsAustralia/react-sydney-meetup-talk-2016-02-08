exports.toHtml = function toHtml(jsxComponent, tagName, fisoBootArguments) {
    var unqiueId  = cuid();
    var reactHtml = ReactDOMServer.renderToStaticMarkup(jsxComponent);

    return `<div id="${unqiueId}">${reactHtml}</div>
    <script>
      window.fisoBoot = window.fisoBoot || {};
      window.fisoBoot['${tagName}'] = window.fisoBoot['${tagName}'] || {};
      window.fisoBoot['${tagName}']['${unqiueId}'] = ${JSON.stringify(fisoBootArguments)};
    </script>`;
};
