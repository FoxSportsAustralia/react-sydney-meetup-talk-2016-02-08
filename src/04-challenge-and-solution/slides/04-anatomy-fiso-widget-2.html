function Widget(element, settings) { // our widget
    this.element = element;
    this.settings = settings;
}

let widget = new Widget(null, {});
widget.initIso();

Widget.prototype.initIso = function () {
    return new Promise((resolve) => {
        this.getData()
            .onValue((data) => {
                resolve({ html: this.render(data) });
            });
    });
};
