Widget.prototype.getData = function (initialData = false) {
    let posts = initalLaterOr(
        initialData && initialData.posts,
        this.getWidgetStream()
    ); // initialData doesn't exist yet so we fetch the stream

    return bacon.combineTemplate({
        view: bacon.combineTemplate({
            posts,
            footerText : this.settings.footerText,
            carousel   : this.settings.carousel
        }),
        iso: bacon.combineTemplate({
            posts
        })
    });
};
